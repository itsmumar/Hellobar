187describe ScriptGenerator, '#rules' do
  
188  let(:site) { double 'site', id: '1337', rules: [], bars: [] }
  
189  let(:generator) { ScriptGenerator.new(site) }
  
190
  
191  it 'returns the proper array of hashes for a sites rules' do
  
192    rule = Rule.new id: 1
  
193    site.stub rules: [rule]
  
194    generator.stub bars_for_rule: []
  
195
  
196    expected_hash = {
  
197      bar_json: [].to_json,
  
198      priority: 1,
  
199      metadata: { "id" => 1 }.to_json,
  
200      rule_eligibility: 'return true;}'
  
201    }
  
202
  
203    generator.rules.should == [expected_hash]
  
204  end
  
205
  
206  it 'returns the proper hash when a single bar_id is passed as an option' do
  
207    rule = Rule.create
  
208    bar = Bar.create bar_type: 'email', rule: rule
  
209    options = { bar_id: bar.id }
  
210
  
211    generator = ScriptGenerator.new(site, options)
  
212    generator.stub bar_settings: {id: bar.id, template_name: bar.bar_type}
  
213
  
214    site.stub rules: [rule]
  
215
  
216    expected_hash = {
  
217      bar_json: [{ id: bar.id, template_name: bar.bar_type }].to_json,
  
218      priority: 1,
  
219      metadata: { "id" => rule.id }.to_json,
  
220      rule_eligibility: 'return true;}'
  
221    }
  
222
  
223    generator.rules.should == [expected_hash]
  
224  end
  
225

229    options = { render_paused_bars: true }
  
230    generator = ScriptGenerator.new(site, options)

265describe ScriptGenerator, '#generate_script' do
  
266  it 'does not compress the template if the compress option is not set' do
  
267    generator = ScriptGenerator.new('site')
  
268    generator.stub :render => 'template'
  
269
  
270    Uglifier.should_not_receive(:new)
  
271    generator.should_receive(:render)
  
272
  
273    generator.generate_script
  
274  end
  
275
  
276  it 'compresses the template when the compress option is true' do
  
277    generator = ScriptGenerator.new('site', { compress: true })
  
278    generator.stub :render => 'template'
  
279
  
280    uglifier = Uglifier.new
  
281    Uglifier.should_receive(:new).and_return(uglifier)
  
282    uglifier.should_receive(:compress).with('template')
  
283
  
284    generator.generate_script
  
285  end
