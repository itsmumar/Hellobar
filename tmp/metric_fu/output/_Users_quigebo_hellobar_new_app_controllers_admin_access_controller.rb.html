7  def do_reset_password
  
8    if current_admin.password_hashed != current_admin.encrypt_password(params[:existing_password])
  
 9      @error = "Your existing password is incorrect"
  
10    elsif params[:new_password] != params[:new_password_again]
  
11      @error = "Your new passwords did not match each other"
  
12    elsif !params[:new_password] or params[:new_password].length < Admin::MIN_PASSWORD_LENGTH
  
13      @error = "New password must be at least #{Admin::MIN_PASSWORD_LENGTH} chars"
  
14    elsif params[:new_password] == params[:existing_password]
  
15      @error = "New password must be different than existing password."
  
16    end
  
17
  
18    if @error
  
19      flash.now[:error] = @error
  
20      render :reset_password
  
21    else
  
22      current_admin.reset_password!(params[:new_password])
  
23      redirect_to admin_path
  
24    end
  
25  end

101    return redirect_to(admin_locked_path) if admin.locked?
  
102
  
103    # If they have validated the access token then we
  
104    # can render step 2
  
105    if admin.has_validated_access_token?(access_token)
  
106      if admin.needs_mobile_code?(access_token)
  
107        admin.send_new_mobile_code!
  
108      end
  
109    else
  
110      admin.send_validate_access_token_email!(access_token)
  
111      return render(:validate_access_token)
  
112    end

114    # Might get locked in the process of sending a new mobile code
  
115    # (exceeds attempts)
