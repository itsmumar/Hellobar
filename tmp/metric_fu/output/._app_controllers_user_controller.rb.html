1class UserController < ApplicationController
  
2  layout "with_sidebar"
  
3
  
4  before_filter :authenticate_user!
  
5  before_filter :load_user, :only => [:edit, :update]
  
6
  
7  def update
  
8    if @user.update_attributes(user_params)
  
 9      sign_in @user, :bypass => true
  
10      flash[:success] = "Your settings have been updated."
  
11      redirect_to current_site ? site_path(current_site) : new_site_path
  
12    else
  
13      flash.now[:error] = "There was a problem updating your settings."
  
14      render :action => :edit
  
15    end
  
16  end
  
17
  
18
  
19  private
  
20
  
21  def load_user
  
22    @user = current_user
  
23  end
  
24
  
25  def user_params
  
26    if params[:user] && params[:user][:password].blank?
  
27      params[:user].delete(:password)
  
28      params[:user].delete(:password_confirmation)
  
29    end
  
30
  
31    params.require(:user).permit(:email, :first_name, :last_name, :password, :password_confirmation)
  
32  end
  
33end
