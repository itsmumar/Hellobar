 9        internal_processing_metadata_set = false
  
10        if last_updated_at
  
11          internal_processing_metadata_set = true
  
12        end

14        # Process the events and props
  
15        last_visitor_user_id_processed = process_visitor_user_ids(last_visitor_user_id_processed)

28      def get_person(target_type, target_id)
  
29        # First see if we have the person in the DB already
  
30        unless person = InternalPerson.where(:"#{target_type}_id"=>target_id).first
  
31          # Nope - so create them
  
32          person = InternalPerson.new
  
33          case target_type
  
34          when "visitor"
  
35            person.visitor_id = target_id
  
36          when "user"
  
37            person.user_id = target_id.to_i
  
38          end
  
39        end
  
40        return person if person
  
41      end

43      def merge_people(person1, person2)
  
44        person = InternalPerson.new
  
45        person.visitor_id = person1.visitor_id || person2.visitor_id
  
46        person.user_id = person1.user_id || person2.user_id
  
47        # Use whichever timestamp is greater
  
48        [:first_visited_at, :signed_up_at, :completed_registration_at, :created_first_bar_at, :created_second_bar_at, :received_data_at].each do |timestamp|
  
49          person.send("#{timestamp}=".to_sym, [person2.send(timestamp), person1.send(timestamp)].compact.max)
  
50        end
  
51        # Update attributes
  
52        cxn = ActiveRecord::Base.connection
  
53        person1_id = person1.id
  
54        person2_id = person2.id
  
55        person1.destroy
  
56        person2.destroy
  
57        person.save
  
58
  
59        dimensions = {}
  
60        to_delete = []
  
61        # Load up all the dimensions for both exsting users
  
62        cxn.execute("SELECT person_id, name, value, timestamp FROM `internal_dimensions` WHERE person_id = #{person1_id} OR person_id = #{person2_id}").each do |row|
  
63          data = {person_id: row[0], name: row[1], value: row[2], timestamp: row[3]}

64          # Only keep one dimension (whichever is later)
  
65          if existing = dimensions[data[:name]]
  
66            if data[:timestamp] > existing[:timestamp]
  
67              to_delete << existing
  
68              dimensions[data[:name]] = data
  
69            else
  
70              to_delete << data
  
71            end
  
72          else
  
73            # Just one so far
  
74            dimensions[data[:name]] = data
  
75          end
  
76        end
  
77        # Update the person of the ones we are keeping
  
78        dimensions.values.each do |row|
  
79          cxn.execute("UPDATE `internal_dimensions` SET person_id=#{person.id} WHERE person_id = #{row[:person_id]} and name = #{cxn.quote(row[:name])}")
  
80        end
  
81        # Delete the ones we are not keeping
  
82        to_delete.each do |row|
  
83          cxn.execute("DELETE FROM `internal_dimensions` WHERE person_id = #{row[:person_id]} and name = #{cxn.quote(row[:name])}")
  
84        end
  
85
  
86        return person
  
87      end

89      def process_visitor_user_ids(last_visitor_user_id_processed)
  
90        cxn = ActiveRecord::Base.connection
  
91        cxn.execute("SELECT * FROM `internal_props` WHERE `name` = 'user_id' AND id > #{last_visitor_user_id_processed || 0}").each do |row|
  
92          id, timestamp, target_type, target_id, name, value = *row
  
93          person_by_visitor = get_person(target_type, target_id)
  
94          person_by_user = get_person("user", value.to_i)
  
95
  
96          method = nil
  
97          if person_by_visitor.new_record? and !person_by_user.new_record?
  
98            # The person by user already exists and we just need to update
  
 99            # the visitor id
  
100            method = :set_visitor_id
  
101            person = person_by_user
  
102          elsif !person_by_visitor.new_record? and person_by_user.new_record?
  
103            # The person by visitor already exists we just need to update
  
104            # the user id
  
105            method = :set_user_id
  
106            person = person_by_visitor
  
107          elsif !person_by_visitor.new_record? and !person_by_user.new_record?
  
108            # They both exist, if they are not the same person we merge them
  
109            if person_by_visitor.id != person_by_user.id
  
110              person = merge_people(person_by_visitor, person_by_user)
  
111            else
  
112              # They are the same person - no need to set anything
  
113              person = person_by_visitor
  
114            end
  
115          elsif person_by_visitor.new_record? and person_by_user.new_record?
  
116            # They are both new records, so lets use the person_by_visitor
  
117            person = person_by_visitor
  
118            method = :set_user_id
  
119          end
  
120
  
121          case method
  
122          when :set_visitor_id
  
123            person.visitor_id = target_id unless person.visitor_id
  
124          when :set_user_id
  
125            person.user_id = value.to_i
  
126          end
  
127
  
128          # Update the first visited at if needed
  
129          person.first_visited_at = timestamp if !person.first_visited_at or timestamp < person.first_visited_at
  
130          # Save any changes
  
131          person.save if person.changed?
  
132
  
133          last_visitor_user_id_processed = id
  
134        end
  
135
  
136        return last_visitor_user_id_processed
  
137      end

139      def process_events(last_event_processed)
  
140        cxn = ActiveRecord::Base.connection
  
141        valid_events = ["Signed Up", "Created First Bar", "Created Second Bar", "Completed Registration", "Received Data"]
  
142        cxn.execute("SELECT * FROM `internal_events` WHERE `name` IN (#{valid_events.collect{|e| cxn.quote(e)}.join(", ")}) AND id > #{last_event_processed || 0}").each do |row|
  
143          id, timestamp, target_type, target_id, name = *row
  
144          person = get_person(target_type, target_id)
  
145          person.first_visited_at = timestamp if !person.first_visited_at or timestamp < person.first_visited_at
  
146          case name
  
147            when "Signed Up"
  
148              person.signed_up_at ||= timestamp
  
149            when "Created First Bar"
  
150              person.created_first_bar_at ||= timestamp
  
151            when "Created Second Bar"
  
152              person.created_second_bar_at ||= timestamp
  
153            when "Completed Registration"
  
154              person.completed_registration_at ||= timestamp
  
155            when "Received Data"
  
156              person.received_data_at ||= timestamp
  
157          end
  
158          if person.changed?
  
159            person.save
  
160          end
  
161          last_event_processed = id
  
162        end
  
163        return last_event_processed
  
164      end

166      def process_props(last_prop_processed)
  
167        cxn = ActiveRecord::Base.connection
  
168        invalid_props = ["user_id"]
  
169        cxn.execute("SELECT * FROM `internal_props` WHERE `name` NOT IN (#{invalid_props.collect{|e| cxn.quote(e)}.join(", ")}) AND id > #{last_prop_processed || 0}").each do |row|
  
170          id, timestamp, target_type, name, value, target_id = *row
  
171          person = get_person(target_type, target_id)
  
172          person.first_visited_at = timestamp if !person.first_visited_at or timestamp < person.first_visited_at
  
173          if person.changed?
  
174            person.save
  
175          end
  
176
  
177          # See if we can find the internal dimension and create it
  
178          # if we can't find it
  
179          attributes = {:person_id=>person.id, :name=>name}
  
180          dimension = InternalDimension.where(attributes).first || InternalDimension.new(attributes)
  
181          # See if we have different value then what was established
  
182          if value != dimension.value and (!dimension.timestamp || timestamp > dimension.timestamp)
  
183            # Update it
  
184            dimension.value = value
  
185            dimension.timestamp = timestamp
  
186            # Need to write your own save code because we
  
187            # are using a primary composite key
  
188            if dimension.new_record?
  
189              cxn.execute("INSERT INTO `internal_dimensions` VALUES(#{dimension.person_id}, #{cxn.quote(dimension.name)}, #{cxn.quote(dimension.value)}, #{dimension.timestamp})")
  
190            else
  
191              cxn.execute("UPDATE `internal_dimensions` SET value = #{cxn.quote(dimension.value)}, timestamp = #{dimension.timestamp} WHERE person_id = #{dimension.person_id} AND name = #{cxn.quote(dimension.name)}")
  
192            end
  
193          end
  
194          last_prop_processed = id
  
195        end
  
196        return last_prop_processed
  
197      end
