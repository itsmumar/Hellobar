3set :application, "hellobar"
  
4set :repo_url, "git@github.com:PolymathicMedia/hellobar_new.git"
  
5set :deploy_to, "/mnt/deploy"
  
6set :linked_files, %w{config/database.yml config/secrets.yml config/settings.yml}
  
7set :linked_dirs, %w{log tmp/pids}
  
8set :rails_env, "production"
  
 9set :ssh_options, { :forward_agent => true }
  
10
  
11# Default branch is :master

15namespace :deploy do
  
16  desc "Restart application"
  
17  task :restart do
  
18    on roles(:web) do
  
19      invoke "deploy:reload_nginx_config"
  
20      invoke "deploy:restart_thin"
  
21      invoke "deploy:restart_monit"
  
22    end
  
23  end
  
24
  
25  after :publishing, :restart
  
26
  
27  task :reload_nginx_config do
  
28    on roles(:web) do
  
29      # uses kill, but actually just reloads the config.
  
30      execute "sudo kill -HUP `cat /mnt/deploy/shared/pids/nginx.pid` || sudo nginx -c /mnt/deploy/current/config/nginx/#{fetch(:stage)}.web.conf"
  
31    end
  
32  end
  
33
  
34  task :restart_thin do
  
35    on roles(:web) do
  
36      execute "cd #{release_path} && sudo bundle exec thin restart -C config/thin/www.yml"
  
37    end
  
38  end
  
39
  
40  task :start_thin do
  
41    on roles(:web) do
  
42      execute "cd #{release_path} && sudo bundle exec thin start -C config/thin/www.yml"
  
43    end
  
44  end
  
45
  
46  task :restart_monit do
  
47    on roles(:web) do
  
48      execute "sudo service monit stop"
  
49      execute "sudo rm /etc/monit/monitrc"
  
50      execute "sudo ln -s /mnt/deploy/current/config/deploy/monitrc /etc/monit/monitrc"
  
51      execute "sudo chown root /etc/monit/monitrc"
  
52      execute "sudo service monit start"
  
53      execute "sudo monit restart all"
  
54    end
  
55  end
  
56end
