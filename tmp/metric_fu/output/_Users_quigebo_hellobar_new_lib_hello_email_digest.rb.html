71    def bar_metrics_for_site(site, bar_data)
  
72      return nil if bar_data.empty?
  
73
  
74      this_week, last_week = bar_data.partition{|bd| bd.segment >= 7.days.ago.strftime("%Y%m%d").to_i}
  
75      metrics = {}
  
76
  
77      # calculate this week's metrics
  
78      metrics[:views] = {:n => this_week.sum(&:views)}
  
79      metrics[:actions] = {:n => this_week.sum(&:conversions)}
  
80      metrics[:conversion] = {:n => this_week.sum(&:views) == 0 ? 0 : (this_week.sum(&:conversions).to_f / this_week.sum(&:views)).round(4)}
  
81
  
82      # calculate lift over last week if we have sufficient data
  
83      if last_week.any? && site.script_installed_at && site.script_installed_at <= 14.days.ago.beginning_of_day
  
84        if last_week.sum(&:views) == 0
  
85          metrics[:views][:lift] = nil
  
86        else
  
87          metrics[:views][:lift] = (metrics[:views][:n] - last_week.sum(&:views)) / last_week.sum(&:views).to_f
  
88        end
  
89
  
90        if last_week.sum(&:conversions) == 0
  
91          metrics[:actions][:lift] = nil
  
92        else
  
93          metrics[:actions][:lift] = (metrics[:actions][:n] - last_week.sum(&:conversions)) / last_week.sum(&:conversions).to_f
  
94        end
  
95
  
96        if metrics[:conversion][:n].nil? || last_week.sum(&:views) == 0 || last_week.sum(&:conversions) == 0
  
97          metrics[:conversion][:lift] = nil
  
98        else
  
 99          conversion = (last_week.sum(&:conversions).to_f / last_week.sum(&:views)).round(4)
  
100          metrics[:conversion][:lift] = ((metrics[:conversion][:n] - conversion) / conversion).round(4)
  
101        end
  
102      end
  
103
  
104      return metrics
  
105    end
